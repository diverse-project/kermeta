
template FSM returns "http://fsm#//FSM" = 
iterate over "http://fsm#//FSM/ownedState" as (call State)
iterate over "http://fsm#//FSM/transition" as (call Transition)
.

template State returns "http://fsm#//State" = 
state feat:"http://fsm#//State/name" ;
.

template Transition returns "http://fsm#//Transition" =
transition from ref:"http://fsm#//State/name" into "http://fsm#//Transition/source" to ref:"http://fsm#//State/name" into "http://fsm#//Transition/target"
input feat:"http://fsm#//Transition/input" output feat:"http://fsm#//Transition/output" ;
.
